Bridge Pattern

Decouple abstraction from implementation so both can evolve independently.

Example (Java)


interface Device {
    void turnOn();
    void turnOff();
}

class TV implements Device {
    public void turnOn() { System.out.println("TV ON"); }
    public void turnOff() { System.out.println("TV OFF"); }
}

abstract class Remote {
    protected Device device;
    public Remote(Device device) {
        this.device = device;
    }
    public abstract void togglePower();
}

class BasicRemote extends Remote {
    public BasicRemote(Device device) {
        super(device);
    }
    public void togglePower() {
        device.turnOn();
    }
}

public class Demo {
    public static void main(String[] args) {
        Device tv = new TV();
        Remote remote = new BasicRemote(tv);
        remote.togglePower();
    }
}


interface IDevice {
    void TurnOn();
    void TurnOff();
}

class TV : IDevice {
    public void TurnOn() => Console.WriteLine("TV ON");
    public void TurnOff() => Console.WriteLine("TV OFF");
}

abstract class Remote {
    protected IDevice device;
    protected Remote(IDevice device) {
        this.device = device;
    }
    public abstract void TogglePower();
}

class AdvancedRemote : Remote {
    public AdvancedRemote(IDevice device) : base(device) {}
    public override void TogglePower() {
        device.TurnOn();
    }
}

class Program {
    static void Main() {
        IDevice tv = new TV();
        Remote remote = new AdvancedRemote(tv);
        remote.TogglePower();
    }
}
